/**
 * Validate version string
 */
export function isValidVersion(version: string): boolean {
  // Matches semver patterns: ^1.0.0, ~2.0.0, 1.2.3, >=1.0.0, etc.
  return /^[\^~>=<]?\d+\.\d+\.\d+(-[a-zA-Z0-9.-]+)?$/.test(version)
}
